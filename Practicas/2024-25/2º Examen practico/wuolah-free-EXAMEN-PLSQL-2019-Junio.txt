create or replace TRIGGER TRIGGER_AGENTE 
BEFORE DELETE OR INSERT OR UPDATE ON AGENTE 
FOR EACH ROW
DECLARE 
ACCIDENTES NUMBER;
BEGIN
    SELECT COUNT(*) INTO ACCIDENTES FROM AGENTE A, ACUDE AC, INCIDENTE I WHERE :NEW.LICENCIA  = AC.LICENCIA1 OR :NEW.LICENCIA = 
    AC.LICENCIA2 AND AC.ID_INCIDENTE  = I.ID_INCIDENTE;
    
    IF ACCIDENTES >6 THEN
        RAISE_APPLICATION_ERROR(-20001, 'ERROR EL AGENTE ' || :NEW.LICENCIA || ' NO PUEDE ACUDIR A MAS INCIDENTES');
    END IF;
END;

create or replace TRIGGER TRIGGER_INCENTIVOS 
AFTER INSERT OR UPDATE ON EQUIPO 
FOR EACH ROW
DECLARE
CICLISTASCONMAILLOT NUMBER;
MAIGANADORES NUMBER;
MAIGANADORESMONTAŃEROS NUMBER;
BEGIN
  SELECT COUNT(*) INTO CICLISTASCONMAILLOT FROM CICLISTA C, DISPUTA D, LLEVA L WHERE :NEW.NOMBRE_EQUIPO = C.NOMBRE_EQUIPO AND C.NOMBRE_CICLISTA = D.NOMBRE_CICLISTA AND D.NOMBRE_CICLISTA = L.NOMBRE_CICLISTA;
 
  IF CICLISTASCONMAILLOT <> 0 THEN
    :NEW.PRESUPUESTO := (:NEW.PRESUPUESTO + 100) * CICLISTASCONMAILLOT;
END IF;

  SELECT COUNT(*) INTO MAIGANADORES FROM DISPUTA D WHERE D.ORDEN_CLASIFICACION = '1' AND D.NOMBRE_CICLISTA IN( SELECT C.NOMBRE_CICLISTA FROM CICLISTA C, DISPUTA D, LLEVA L WHERE :NEW.NOMBRE_EQUIPO = C.NOMBRE_EQUIPO AND C.NOMBRE_CICLISTA = D.NOMBRE_CICLISTA AND D.NOMBRE_CICLISTA = L.NOMBRE_CICLISTA);
  
    IF MAIGANADORES <> 0 THEN
    :NEW.PRESUPUESTO := (:NEW.PRESUPUESTO + 100) * MAIGANADORES;
END IF;

    SELECT COUNT(*) INTO MAIGANADORESMONTAŃEROS  FROM ETAPAMONTANA EM , DISPUTA D, CICLISTA C, LLEVA L   WHERE :NEW.NOMBRE_EQUIPO := C.NOMBRE_EQUIPO    AND  D.ORDEN_CLASIFICACION = '1'     AND C.NOMBRE_CICLISTA = D.NOMBRE_CICLISTA     AND D.NOMBRE_CICLISTA = L.NOMBRE_CICLISTA     AND EM.NUM_ETAPA = D.NUM_ETAPA;
  
    IF MAIGANADORESMONTAŃEROS <> 0 THEN
    :NEW.PRESUPUESTO := (:NEW.PRESUPUESTO + 100) * MAIGANADORESMONTAŃEROS;
END IF;

END;

create or replace PROCEDURE PROCEIMIENTO_EJERCICIO_2 AS 
CURSOR CICLISTA IS SELECT DISTINCT C.NOMBRE_CICLISTA, C.NOMBRE_EQUIPO, E.PRESUPUESTO, ES.CANTIDAD FROM CICLISTA C, EQUIPO E, ES_SUBVENCIONADO ES, DISPUTA D, ETAPAMONTANA EM, ETAPA ET WHERE C.NOMBRE_EQUIPO = E.NOMBRE_EQUIPO AND E.NOMBRE_EQUIPO = ES.NOMBRE_EQUIPO AND C.NOMBRE_CICLISTA = D.NOMBRE_CICLISTA AND D.NUM_ETAPA = ET.NUM_ETAPA AND ET.NUM_ETAPA = EM.NUM_ETAPA;
NPATROCINADORES NUMBER;
CURSOR etapas(CICLISTA VARCHAR2) IS 
SELECT  DISTINCT EM.NUM_ETAPA, EM.ALTITUD, L.ID_MAILLOT, D.ORDEN_CLASIFICACION FROM DISPUTA D, ETAPAMONTANA EM , ETAPA E, LLEVA L
WHERE D.NOMBRE_CICLISTA = CICLISTA AND D.NUM_ETAPA = E.NUM_ETAPA AND E.NUM_ETAPA  = EM.NUM_ETAPA AND L.NOMBRE_CICLISTA = CICLISTA;
ORDEN NUMBER;
MAILLOTS CHAR;
PRIMEROS VARCHAR2(800);
MAILLOT VARCHAR2(800);
BEGIN
    FOR C IN CICLISTA LOOP
        SELECT COUNT(*) INTO NPATROCINADORES FROM PATROCINADOR P, ES_SUBVENCIONADO ES, EQUIPO E 
        WHERE E.NOMBRE_EQUIPO = C.NOMBRE_EQUIPO AND ES.NOMBRE_EQUIPO = E.NOMBRE_EQUIPO AND ES.ID_PATROCINADOR = P.ID_PATROCINADOR;
        DBMS_OUTPUT.PUT_LINE('CICLISTA: '||C.NOMBRE_CICLISTA||' '||C.NOMBRE_EQUIPO||' '|| C.PRESUPUESTO ||' '||NPATROCINADORES||' '||C.CANTIDAD);
        FOR R IN ETAPAS(C.NOMBRE_CICLISTA) LOOP
            IF R.ORDEN_CLASIFICACION <= 4  THEN
                PRIMEROS := 'SI';
            ELSE 
                PRIMEROS := 'NO';
            END IF;
            IF R.ID_MAILLOT <> 0 THEN
                MAILLOT := 'SI';
            ELSE 
                MAILLOT := 'NO';
            END IF;
            DBMS_OUTPUT.PUT_LINE('---> ETAPA MONTAŃA: '||R.NUM_ETAPA||' '||R.ALTITUD||' '||PRIMEROS||' '||MAILLOT);
        END LOOP;
    END LOOP;
END PROCEIMIENTO_EJERCICIO_2;

create or replace FUNCTION FUNCION_EJERCICIO_3 
(
  ID_INCIDENTE IN VARCHAR2 
, ID_CARRETERA OUT VARCHAR2
, ORDEN_CARRETERA OUT VARCHAR2
) RETURN NUMBER AS 
NUM_TM NUMBER :=0;
BEGIN
    SELECT I.ID_CARRETERA, I.ORDEN_CARRETERA, COUNT(*) INTO ID_CARRETERA, ORDEN_CARRETERA, NUM_TM 
    FROM INCIDENTE I,  TERMINOMUNICIPAL TM, PATROCINADOR P , PASA PA
    WHERE ID_INCIDENTE  = I.ID_INCIDENTE 
    AND PA.ORDEN_CARRETERA = I.ORDEN_CARRETERA
    AND PA.ID_TM = TM.ID_TM
    AND TM.ID_TM = P.ID_TM
    AND P.ID_TM = (SELECT ID_TM,  MAX(SUM(ID_TM)) FROM PATROCINADOR GROUP BY ID_TM) group by I.ID_CARRETERA, I.ORDEN_CARRETERA;
  RETURN NUM_TM;
END FUNCION_EJERCICIO_3;
